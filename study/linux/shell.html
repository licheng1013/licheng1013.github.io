IP=`ifconfig $netName | awk 'NR==2{print $2}'

# 替换某个文件的内容

# 追加指令输入
mysql -uroot -p$password <<EOF
USE mysql;
update user set Host="%" where User="root";
select User,authentication_string,Host from user;
flush privileges;
flush privileges;
EOF
