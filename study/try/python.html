def get_img(html_url=home_url+image_suffix, image_url=home_url):
    str_html = requests.get(html_url)
    soup = BeautifulSoup(str_html.text, 'lxml')
    img_list = []
    for i in range(1, 30):
        # éœ€è¦è·å–æ ‡ç­¾
        _str = soup.select('#main > div.slist > ul > li:nth-child(' + str(i) + ') > a > img')

            img = str(_str[0])
            start = img.find("src")
            end = img.find(".jpg")

    return img_list
