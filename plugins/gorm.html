                              # 功能

                              • 1: 支持对gorm框架的联动数据库表进行实体生成
                              • 2: 支持对gorm+gin框架的联动数据库进行接口与实体生成
                              • 3: 支持生成json tag 和 form tag
                              • 4: 支持根据字段或者注释自动映射到gorm框架中的联动属性，自动创建和修改时间
                              • 5: 支持根据字段描述来联动乐观锁

                              package demo

import (
	"gin-gin/common"
	"gin-gin/config"
	"gin-gin/entity"
	"github.com/gin-gonic/gin"
	"gorm.io/gorm"
	"log"
	"net/http"
)

func init() {
	c := &TestController{}
	c.init(config.R)
}

func (t TestController) init(g *gin.Engine) {
	// 依次: 分页列表，单条，插入，修改，删除
	group := g.Group("test")
	group.GET("/list", t.list) //不设置限制条件的画默认查询所有
	group.GET("/one", t.one)
	group.POST("/insert", t.insert)
	group.POST("/update", t.update)
	group.POST("/delete", t.delete)
}

// TestController 控制器
type TestController struct {
	Page int   `form:"page"`
	Size int   `form:"size"`
	Ids  []int `form:"ids"`
}

func (t TestController) db() *gorm.DB {
	return common.Db
}

// 分页列表
func (t TestController) list(c *gin.Context) {
	_ = c.Bind(&t)
	v := &Test{}
	_ = c.Bind(&v)

	// 结果
	var lists []Test
	t.db().Model(&v).Where(&v).Order("").Offset((t.Page - 1) * t.Size).Limit(t.Size).Find(&lists)
	// 统计
	var total int64
	t.db().Model(&v).Where(&v).Count(&total)

	data := make(map[string]interface{})
	data["list"] = lists
	data["total"] = total
	c.JSON(http.StatusOK, entity.OkData(data))
}

// 根据主键Id查询记录
func (t TestController) one(c *gin.Context) {
	var v Test
	_ = c.Bind(&v)
	db := t.db().Find(&v, v.Id)
	if db.RowsAffected != 1 {
		log.Println("未找到数据！")
	}
	c.JSON(http.StatusOK, entity.OkData(v))
}

// 修改记录
func (t TestController) update(c *gin.Context) {
	var v Test
	_ = c.ShouldBindJSON(&v)
	tx := t.db().Model(&v).Updates(v)
	if tx.Error != nil {
		log.Panicln(tx.Error.Error())
	}
	c.JSON(http.StatusOK, entity.OkMsg("修改成功！"))
}

// 插入记录
func (t TestController) insert(c *gin.Context) {
	var v Test
	_ = c.ShouldBindJSON(&v)
	tx := t.db().Save(&v)
	if tx.Error != nil {
		log.Panicln(tx.Error.Error())
	}
	c.JSON(http.StatusOK, entity.OkMsg("插入成功！"))
}

// 根据主键删除
func (t TestController) delete(c *gin.Context) {
	_ = c.Bind(&t)
	tx := t.db().Delete(&Test{}, t.Ids)
	if tx.Error != nil {
		log.Panicln(tx.Error.Error())
	}
	c.JSON(http.StatusOK, entity.OkMsg("删除成功！"))
}
