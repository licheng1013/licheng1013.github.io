                                  # 配置相对路径

                                  • electron 里面需要引入vue文件需要加上.vue
                                  • vite.config.js
                                  import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'
import {join} from 'path'
// https://vitejs.dev/config/
export default defineConfig({
  resolve: {
    alias: {
      '@':join(__dirname,'src')
    }
  },
  plugins: [vue()]
})
                                  • tsconfig.json
                                  {
  "compilerOptions": {
    "baseUrl": "./",
    "paths": {
      "@/*": ["src/*"]
    }
  }
}
